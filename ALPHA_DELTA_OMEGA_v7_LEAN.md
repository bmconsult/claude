## Part 4: Quality Gates (Executable)

```python
def quality_gate(phase: str, output: str) -> tuple[bool, list[str]]:
    """Returns (passed, failed_checks)"""

    if phase == "GENESIS":
        checks = {
            "4 approaches": output.count("### Approach") >= 4,
            "Top pick justified": "Top Pick:" in output and "Why:" in output,
            "Dependencies listed": "MUST be true:" in output,
            "Betting test": "Would I bet $10,000" in output,
            "Confidence stated": "Overall GENESIS confidence:" in output,
        }

    elif phase == "LOGOS":
        checks = {
            "Argument form": "Premise 1:" in output,
            "Inference table": "| Inference |" in output,
            "Verification status": "VERIFIED" in output or "UNVERIFIED" in output,
            "Betting test": "Would I bet $10,000" in output,
            "Confidence stated": "Overall LOGOS confidence:" in output,
        }

    elif phase == "ADVERSARY":
        checks = {
            "All 12 attacks": all(f"Δ.{i}" in output or f"ATTACK {i}" in output for i in range(1, 13)),
            "Steelman present": "Steelman" in output and len(output.split("Steelman")[1].split("##")[0]) > 200,
            "Falsification criteria": "FALSE if:" in output,
            "Calibration box": "CALIBRATION RESULT" in output,
            "Attack verdict": any(v in output for v in ["DESTROYED", "WOUNDED", "SURVIVES"]),
        }

    failed = [name for name, passed in checks.items() if not passed]
    return len(failed) == 0, failed
```
