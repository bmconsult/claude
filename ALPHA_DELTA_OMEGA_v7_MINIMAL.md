def quality_gate(phase: str, output: str) -> tuple[bool, list[str]]:
    """Returns (passed, failed_checks)"""

    if phase == "CREATE":
        checks = {
            "4 approaches": output.count("Approach") >= 4,
            "Selection justified": "Selected:" in output and "Why" in output,
            "Logical structure": "Premise 1:" in output,
            "Dependencies": "MUST be true:" in output,
            "Betting test": "Would I bet $10,000" in output,
            "Confidence": "CREATE confidence:" in output,
        }

    elif phase == "TEST":
        checks = {
            "12 attacks": sum(1 for i in range(1, 13) if f"Δ.{i}" in output) >= 10,
            "Steelman devastating": "Steelman" in output and "DEVASTATING" in output.upper() or len(output.split("Steelman")[1].split("##")[0]) > 300,
            "Falsification specific": "WRONG if:" in output,
            "Calibration complete": all(x in output for x in ["Γ", "Ε", "Μ"]),
            "Verdict present": any(v in output for v in ["DESTROYED", "WOUNDED", "SURVIVES"]),
            "Betting test": "bet $10K" in output,
        }

    failed = [name for name, passed in checks.items() if not passed]
    return len(failed) == 0, failed
